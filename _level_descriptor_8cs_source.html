237 public bool[] GetTraversibles(){
238 var traversibles = new bool[Width()*Height()];
239 //Init them all to be true
240 for (int i = 0; i < traversibles.Length; i++)
241 {
242 traversibles[i] = true;
243 }
244 //Check against bases
245 for (int y = 0; y < layoutDescriptor.levelLayout.rows.Length; y++)
246 {
247 for (int x = 0; x < layoutDescriptor.levelLayout.rows[y].tiles.Length; x++)
248 {
249 int index = y * Width() + x;
250 //Get the required base asset
251 if (this._LevelDescriptorCollection._BaseTileDB.GetFromList(layoutDescriptor.levelLayout.rows[y].tiles[x].baseTileDescriptor.name, out BaseTileAsset baseTileAsset))
252 {
253 if (!baseTileAsset.traversible)
254 {
255 traversibles[index] = false;
256 }
257 else
258 {
259 traversibles[index] = true;
260 }
261 }
262 }
263 }
264
265 //Check against extras
266 foreach (var extraDescriptor in layoutDescriptor.levelLayout.extras)
267 {
268 //Get the required extra asset
269 if (this._LevelDescriptorCollection._ExtraTileDB.GetFromList(extraDescriptor.name, out ExtraTileAsset requiredAsset))
270 {
271 var new_coord = TileCoordinate.IndexToCoord(extraDescriptor.index, Width(), Height());
272 var obscuring = requiredAsset.GetObscuringFrom(new_coord, Width());
273 foreach (var obscure in obscuring)
274 {
275 // Debug.Log("Obscuring " + obscure);
276 traversibles[obscure] = false;
277 }
278 }
279 }
280 return traversibles;

293 public void ApplyInsights(InsightResponse insights)
294 {
295 //Need to splice the insights into the characters
296 var patrolInsights = insights.patrols;
297 foreach (var patrolInsight in patrolInsights)
298 {
299 //Find the character by that name
300 if (!Array.Exists(this.characters, (descriptor => descriptor.name == patrolInsight.name)))
301 {
302 Debug.LogError($"Character {patrolInsight.name} does not exist in descriptor");
303 }
304 int index =Array.FindIndex(this.characters, (descriptor => descriptor.name == patrolInsight.name));
305
306 //In that descriptor, get the behavior traits starting with patrol_target
307 var temp_character_descriptor = characters[index];
308 for (int i = 0; i < patrolInsight.patrol_path.Length; i++)
309 {
310 var old_value = temp_character_descriptor.GetBehaviorTarget($"patrol_path_{i + 1}").value;
311 var new_value = patrolInsight.patrol_path[i].ToString();
312 //They start from 1 so
313 temp_character_descriptor.SetBehaviorTarget($"patrol_path_{i+1}", patrolInsight.patrol_path[i].ToString());
314 var new_old_value = temp_character_descriptor.GetBehaviorTarget($"patrol_path_{i + 1}").value;
315 Debug.Log($"Changing patrol target {i + 1} from {old_value} to {new_value}, now {new_old_value}");
316
317 }
318
319 characters[index] = temp_character_descriptor;
320 }
