376public class DirectiveFactory
377 {
378 private List<(string name, Type directive)> directivesByName;
379
380 public DirectiveFactory()
381 {
382 var directiveTypes = Assembly.GetAssembly(typeof(Directive)).GetTypes()
383 .Where(someType=> someType.IsClass && (!someType.IsAbstract) && (someType.IsSubclassOf(typeof(Directive))));
384 directivesByName = new();
385
386 foreach (var directiveType in directiveTypes)
387 {
388 var temp = Activator.CreateInstance(directiveType) as Directive;
389 directivesByName.Add((temp.Name, directiveType));
390 }
391 }
392
393 [CanBeNull]
394 public Directive GetDirective(string directiveType)
395 {
396 var count = directivesByName.Count(entry => entry.name == directiveType);
397 if (count!=1)
398 {
399
400 // var sb = new StringBuilder();
401 Debug.LogError("Name: "+ directiveType + " ,Count: " + count);
402 foreach (var (name, type) in directivesByName)
403 {
404 Debug.Log(name + ", " + type.ToString());
405 }
406 return null;
407 }
408
409 var requiredType = directivesByName.Find((entry) => entry.name == directiveType).directive;
410 var directive = Activator.CreateInstance(requiredType) as Directive;
411 return directive;
412
413 }
