372public class DirectiveFactory
373 {
374 private List<(string name, Type directive)> directivesByName;
375
376 public DirectiveFactory()
377 {
378 var directiveTypes = Assembly.GetAssembly(typeof(Directive)).GetTypes()
379 .Where(someType=> someType.IsClass && (!someType.IsAbstract) && (someType.IsSubclassOf(typeof(Directive))));
380 directivesByName = new();
381
382 foreach (var directiveType in directiveTypes)
383 {
384 var temp = Activator.CreateInstance(directiveType) as Directive;
385 directivesByName.Add((temp.Name, directiveType));
386 }
387 }
388
389 [CanBeNull]
390 public Directive GetDirective(string directiveType)
391 {
392 var count = directivesByName.Count(entry => entry.name == directiveType);
393 if (count!=1)
394 {
395
396 // var sb = new StringBuilder();
397 Debug.LogError("Name: "+ directiveType + " ,Count: " + count);
398 foreach (var (name, type) in directivesByName)
399 {
400 Debug.Log(name + ", " + type.ToString());
401 }
402 return null;
403 }
404
405 var requiredType = directivesByName.Find((entry) => entry.name == directiveType).directive;
406 var directive = Activator.CreateInstance(requiredType) as Directive;
407 return directive;
408
409 }
