323 public void DetectOtherCharacter(Character other, float dotProduct, bool amISubject = false)
324 {
325 Debug.Log($"{Invoker.name} detecting {other.name}");
326 //First, distance factor
327 var distance = Level.Level.instance.ManhattanDistance(Invoker.MyTile.tileKey, other.MyTile.tileKey);
328 //Avoid enemies too far away
329 if (distance > maxDistance)
330 {
331 return;
332 }
333 //Check if they are not in my own faction
334 if (other.faction == Invoker.faction)
335 {
336 Debug.Log($"{other.name} is the same faction as me {Invoker.name}");
337 return;
338 }
339
340 //Take abs of the dotProduct
341 var angleFactor = Mathf.Abs(dotProduct);
342 // Calculate a detection factor
343 var detection_factor = _dotToDetection.Evaluate(angleFactor);
344 var distance_factor = _dotToDetection.Evaluate((distance * 1.0f / maxDistance));
345
346 //Check if it exists
347 if (!_detections.Any(detection => detection.character == other))
348 {
349 var new_detection = new DetectionStruct();
350 // detection.detectionLevel = detection_factor;
351 new_detection.firstDetectedOnTurn = Get0orTurnNumber();
352 // detection.lastSeenOnTurn = Get0orTurnNumber();
353 new_detection.character = other;
354 new_detection.lastSeenOnTile = other.MyTile.tileKey;
355 new_detection.firstDetectedOnTile = other.MyTile.tileKey;
356 _detections.Add(new_detection);
357 }
358 //Get the entry
359 var required_index = _detections.FindIndex(detection => detection.character == other);
360 var detection = _detections[required_index];
361 var old_detection_level = detection.detectionLevel;
362 var new_detection_level = detection.detectionLevel + detection_factor;
363 detection.detectionLevel = new_detection_level;
364 detection.lastSeenOnTurn = Get0orTurnNumber();
365 detection.lastSeenOnTile = other.MyTile.tileKey;
366 _detections[required_index] = detection;
367 Debug.Log($"New detection level {new_detection_level}");
368 if (InvestigateThreshold(old_detection_level, new_detection_level))
369 {
370 _investigating = true;
371 // If not already chasing, set to other target
372 if (!_chasing)
373 {
374 _currentTarget = other;
375 }
376
377 }
378 if (ChaseThreshold(old_detection_level, new_detection_level))
379 {
380 _investigating = false;
381 _chasing = true;
382 if (!_currentTarget)
383 {
384 _currentTarget = other;
385 }
386 HandleChaseTrigger(amISubject);
387 }
