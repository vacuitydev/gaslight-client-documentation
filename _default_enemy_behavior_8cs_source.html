319 public void DetectOtherCharacter(Character other, float dotProduct, bool amISubject = false)
320 {
321if(_debugDetection) Debug.Log($"{Invoker.name} detecting {other.name}");
322 //First, distance factor
323 var distance = Level.Level.instance.ManhattanDistance(Invoker.MyTile.tileKey, other.MyTile.tileKey);
324 //Avoid enemies too far away
325 if (distance > maxDistance)
326 {
327 return;
328 }
329 //Check if they are not in my own faction
330 if (other.faction == Invoker.faction)
331 {
332 if(_debugDetection) Debug.Log($"{other.name} is the same faction as me {Invoker.name}");
333 return;
334 }
335
336 //Take abs of the dotProduct
337 var angleFactor = Mathf.Abs(dotProduct);
338 // Calculate a detection factor
339 var detection_factor = _dotToDetection.Evaluate(angleFactor);
340 var distance_factor = _dotToDetection.Evaluate((distance * 1.0f / maxDistance));
341
342 //Check if it exists
343 if (!_detections.Any(detection => detection.character == other))
344 {
345 var new_detection = new DetectionStruct();
346 // detection.detectionLevel = detection_factor;
347 new_detection.firstDetectedOnTurn = Get0orTurnNumber();
348 // detection.lastSeenOnTurn = Get0orTurnNumber();
349 new_detection.character = other;
350 new_detection.lastSeenOnTile = other.MyTile.tileKey;
351 new_detection.firstDetectedOnTile = other.MyTile.tileKey;
352 _detections.Add(new_detection);
353 }
354 //Get the entry
355 var required_index = _detections.FindIndex(detection => detection.character == other);
356 var detection = _detections[required_index];
357 var old_detection_level = detection.detectionLevel;
358 var new_detection_level = detection.detectionLevel + detection_factor;
359 detection.detectionLevel = new_detection_level;
360 detection.lastSeenOnTurn = Get0orTurnNumber();
361 detection.lastSeenOnTile = other.MyTile.tileKey;
362 _detections[required_index] = detection;
363if(_debugDetection) Debug.Log($"New detection level {new_detection_level}");
364 if (InvestigateThreshold(old_detection_level, new_detection_level))
365 {
366 _investigating = true;
367 // If not already chasing, set to other target
368 if (!_chasing)
369 {
370 _currentTarget = other;
371 }
372
373 }
374 if (ChaseThreshold(old_detection_level, new_detection_level))
375 {
376 _investigating = false;
377 _chasing = true;
378 if (!_currentTarget)
379 {
380 _currentTarget = other;
381 }
382 HandleChaseTrigger(amISubject);
383 }
