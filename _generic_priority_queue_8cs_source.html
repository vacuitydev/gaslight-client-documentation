161 private void CascadeUp(TItem node)
162 {
163 //aka Heapify-up
164 int parent;
165 if (node.QueueIndex > 1)
166 {
167 parent = node.QueueIndex >> 1;
168 TItem parentNode = _nodes[parent];
169 if(HasHigherPriority(parentNode, node))
170 return;
171
172 //Node has lower priority value, so move parent down the heap to make room
173 _nodes[node.QueueIndex] = parentNode;
174 parentNode.QueueIndex = node.QueueIndex;
175
176 node.QueueIndex = parent;
177 }
178 else
179 {
180 return;
181 }
182 while(parent > 1)
183 {
184 parent >>= 1;
185 TItem parentNode = _nodes[parent];
186 if(HasHigherPriority(parentNode, node))
187 break;
188
189 //Node has lower priority value, so move parent down the heap to make room
190 _nodes[node.QueueIndex] = parentNode;
191 parentNode.QueueIndex = node.QueueIndex;
192
193 node.QueueIndex = parent;
194 }
195 _nodes[node.QueueIndex] = node;
196 }

201 private void CascadeDown(TItem node)
202 {
203 //aka Heapify-down
204 int finalQueueIndex = node.QueueIndex;
205 int childLeftIndex = 2 * finalQueueIndex;
206
207 // If leaf node, we're done
208 if(childLeftIndex > _numNodes)
209 {
210 return;
211 }
212
213 // Check if the left-child is higher-priority than the current node
214 int childRightIndex = childLeftIndex + 1;
215 TItem childLeft = _nodes[childLeftIndex];
216 if(HasHigherPriority(childLeft, node))
217 {
218 // Check if there is a right child. If not, swap and finish.
219 if(childRightIndex > _numNodes)
220 {
221 node.QueueIndex = childLeftIndex;
222 childLeft.QueueIndex = finalQueueIndex;
223 _nodes[finalQueueIndex] = childLeft;
224 _nodes[childLeftIndex] = node;
225 return;
226 }
227 // Check if the left-child is higher-priority than the right-child
228 TItem childRight = _nodes[childRightIndex];
229 if(HasHigherPriority(childLeft, childRight))
230 {
231 // left is highest, move it up and continue
232 childLeft.QueueIndex = finalQueueIndex;
233 _nodes[finalQueueIndex] = childLeft;
234 finalQueueIndex = childLeftIndex;
235 }
236 else
237 {
238 // right is even higher, move it up and continue
239 childRight.QueueIndex = finalQueueIndex;
240 _nodes[finalQueueIndex] = childRight;
241 finalQueueIndex = childRightIndex;
242 }
243 }
244 // Not swapping with left-child, does right-child exist?
245 else if(childRightIndex > _numNodes)
246 {
247 return;
248 }
249 else
250 {
251 // Check if the right-child is higher-priority than the current node
252 TItem childRight = _nodes[childRightIndex];
253 if(HasHigherPriority(childRight, node))
254 {
255 childRight.QueueIndex = finalQueueIndex;
256 _nodes[finalQueueIndex] = childRight;
257 finalQueueIndex = childRightIndex;
258 }
259 // Neither child is higher-priority than current, so finish and stop.
260 else
261 {
262 return;
263 }
264 }
265
266 while(true)
267 {
268 childLeftIndex = 2 * finalQueueIndex;
269
270 // If leaf node, we're done
271 if(childLeftIndex > _numNodes)
272 {
273 node.QueueIndex = finalQueueIndex;
274 _nodes[finalQueueIndex] = node;
275 break;
276 }
277
278 // Check if the left-child is higher-priority than the current node
279 childRightIndex = childLeftIndex + 1;
280 childLeft = _nodes[childLeftIndex];
281 if(HasHigherPriority(childLeft, node))
282 {
283 // Check if there is a right child. If not, swap and finish.
284 if(childRightIndex > _numNodes)
285 {
286 node.QueueIndex = childLeftIndex;
287 childLeft.QueueIndex = finalQueueIndex;
288 _nodes[finalQueueIndex] = childLeft;
289 _nodes[childLeftIndex] = node;
290 break;
291 }
292 // Check if the left-child is higher-priority than the right-child
293 TItem childRight = _nodes[childRightIndex];
294 if(HasHigherPriority(childLeft, childRight))
295 {
296 // left is highest, move it up and continue
297 childLeft.QueueIndex = finalQueueIndex;
298 _nodes[finalQueueIndex] = childLeft;
299 finalQueueIndex = childLeftIndex;
300 }

302 {

304 childRight.QueueIndex = finalQueueIndex;

358 public TItem Dequeue()
359 {
360#if DEBUG
361 if(_numNodes <= 0)
362 {
363 throw new InvalidOperationException("Cannot call Dequeue() on an empty queue");
364 }
365
366 if(!IsValidQueue())
367 {
368 throw new InvalidOperationException("Queue has been corrupted (Did you update a node priority manually instead of calling UpdatePriority()?" +
369 "Or add the same node to two different queues?)");
370 }
371#endif
372
373 TItem returnMe = _nodes[1];
374 //If the node is already the last node, we can remove it immediately
375 if(_numNodes == 1)
376 {
377 _nodes[1] = null;
378 _numNodes = 0;
379 return returnMe;
380 }
381
382 //Swap the node with the last node
383 TItem formerLastNode = _nodes[_numNodes];
384 _nodes[1] = formerLastNode;
385 formerLastNode.QueueIndex = 1;
386 _nodes[_numNodes] = null;
387 _numNodes--;
388
389 //Now bubble formerLastNode (which is no longer the last node) down
390 CascadeDown(formerLastNode);
391 return returnMe;

583 {

585 {
